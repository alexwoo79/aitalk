set.seed(2024)

sales_data <- expand.grid(
  date     = seq(as.Date("2024-10-01"), as.Date("2024-12-31"), by = "day"),
  region   = c("华东", "华南", "华北", "西南"),
  category = c("手机", "电脑", "配件", "智能穿戴")
) %>%
  mutate(
    sales_amount = round(runif(n(), 
                               min = ifelse(category == "手机", 5000, 500),
                               max = ifelse(category == "手机", 50000, 8000)), 2),
    quantity     = sample(1:20, n(), replace = TRUE),
    profit       = sales_amount * runif(n(), 0.08, 0.35),
    month        = format(date, "%Y-%m")
  )

2.2 月度汇总对比

Code 
sales_data %>%
  group_by(month) %>%
  summarise(
    总销售额 = sum(sales_amount) / 10000,
    订单数   = n(),
    .groups  = "drop"
  ) %>%
  mutate(
    环比增长 = scales::percent(
      (总销售额 / lag(总销售额) - 1), accuracy = 0.1
    ),
    环比增长 = ifelse(is.na(环比增长), "—", 环比增长)
  ) %>%
  flextable() %>%
  set_header_labels(
    `month`    = "月份",
    `总销售额` = "总销售额（万元）",
    `订单数`   = "订单数",
    `环比增长` = "环比增长"
  ) %>%
  theme_booktabs() %>%
  bg(part = "header", bg = COLOR_PRIMARY) %>%
  color(part = "header", color = "white") %>%
  bold(j = "环比增长") %>%
  color(j = "环比增长", color = COLOR_SUCCESS) %>%
  colformat_double(j = "总销售额", digits = 1) %>%
  align(align = "center", part = "all") %>%
  autofit()

3.1 区域销售对比

region_summary <- sales_data %>%
  group_by(region) %>%
  summarise(
    销售额 = sum(sales_amount) / 10000,
    订单数 = n(),
    利润率 = sum(profit) / sum(sales_amount),
    .groups = "drop"
  ) %>%
  mutate(
    销售额占比 = scales::percent(销售额 / sum(销售额), accuracy = 0.1)
  )

ggplot(region_summary, aes(reorder(region, -销售额), 销售额, fill = region)) +
  geom_col(width = 0.6) +
  geom_text(aes(label = paste0(round(销售额, 1), "万")), 
            vjust = -0.5, size = 4.5, fontface = "bold") +
  scale_fill_manual(values = c("#3498db", "#e74c3c", "#2ecc71", "#f39c12")) +
  labs(
    title    = "各区域销售额对比",
    subtitle = paste("报告期：", params$report_period),
    x = "", y = "销售额（万元）"
  ) +
  theme_minimal(base_size = 13) +

4.1 品类结构

Code 
cat_data <- sales_data %>%
  group_by(category) %>%
  summarise(销售额 = sum(sales_amount) / 10000, .groups = "drop") %>%
  mutate(
    label = paste0(category, "\n", round(销售额 / sum(销售额) * 100, 1), "%")
  )

ggplot(cat_data, aes(x = "", y = 销售额, fill = category)) +
  geom_col(width = 1) +
  coord_polar(theta = "y") +
  geom_text(aes(label = label), position = position_stack(vjust = 0.5), 
            size = 4, color = "white", fontface = "bold") +
  scale_fill_brewer(palette = "Set2") +
  labs(title = "品类销售额占比") +
  theme_void(base_size = 14) +
  theme(
    plot.title = element_text(face = "bold", hjust = 0.5, size = 16),
    legend.position = "none"
  )
